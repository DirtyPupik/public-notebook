function status() {
    if  [ "$#" -ne 1 ] ; then
        echo "usage: status <service name>";
        return 1;
    fi
    service $1 status;
}

function sstart() {
    if  [ "$#" -ne 1 ] ; then
        echo "usage: sstart <service name>   -   will start service and enable";
        return 1;
    fi
    service $1 start && systemctl enable $1;
}

function sstop() {
    if  [ "$#" -ne 1 ] ; then
        echo "usage: sstop <service name>   -   will stop service and disable";
        return 1;
    fi
    service $1 stop && systemctl disable $1;
}

function timestamp() {
    if ! ([ "$#" -eq 1 ] | [[ $1 =~ ^-?[0-9]+$ ]]); then
        echo "usage: timestamp <time stamp>";
        return 1;
    fi
    python3 -c "from datetime import datetime; print(datetime.fromtimestamp($1))";
}
